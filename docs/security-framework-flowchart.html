1. Fluxo de Autenticação (Login)

🌐 Browser
(Next.js)
→
POST /api/auth/login { email, password }
→
⚡ NestJS API
1 LoginThrottlerGuard - Rate limiting: 5 tentativas/min com lockout progressivo
2 AuthService.login() - Valida credenciais, gera JWT accessToken + refreshToken
3 Set Cookies - Tokens em HttpOnly cookies (invisíveis ao JS)
✅ Response: Set-Cookie: accessToken=xxx; HttpOnly; Secure; SameSite=strict
Body: { user, tenant, tenants } ← SEM TOKENS NO JSON!

2. Security Layers (Ordem de Execução)

1 Helmet - Security Headers: CSP, HSTS, X-Frame-Options, nosniff
2 ThrottlerGuard - Rate Limiting Global (100 req/min)
3 JwtAuthGuard - Extrai token do cookie, valida JWT
4 JwtStrategy - Cache 30s, valida user/tenant no DB
5 TenantInterceptor - Injeta tenantId no request
6 TenantGuard - Verifica membership do tenant

3. Matriz de Proteções

import { Request } from 'express'; import { COOKIE_NAMES } from '../../../common/security'; /** * Extrai JWT com prioridade: * 1. HttpOnly cookie (seguro, preferido para browsers) * 2. import { Throttle } from '@nestjs/throttler'; import { LoginThrottlerGuard } from '../../common/guards/login-throttler.guard'; import { getTokenCookieConfig, COOKIE_NAMES } from '../../common/security'; @ApiTags('auth') @Controller('auth') export class AuthController { private readonly cookieConfig: TokenCookieConfig; constructor(private readonly authService: AuthService, configService: ConfigService) { this.cookieConfig = getTokenCookieConfig(configService); } // Seta ambos tokens em HttpOnly cookies private setAuthCookies(res: Response, accessToken: string, refreshToken: string): void { res.cookie(COOKIE_NAMES.ACCESS_TOKEN, accessToken, this.cookieConfig.accessToken); res.cookie(COOKIE_NAMES.REFRESH_TOKEN, refreshToken, this.cookieConfig.refreshToken); } @Public() @Post('register') @Throttle({ default: { limit: 3, ttl: 60000 } }) // 3 registros/min async register(@Body() dto: RegisterDto, @Res({ passthrough: true }) res: Response) { const result = await this.authService.register(dto); } @Public() @Post('login') @UseGuards(LoginThrottlerGuard) // Rate limiting específico para login async login(@Body() dto: LoginDto, @Res({ passthrough: true }) res: Response) { const result = await this.authService.login(dto); return { user: result.user, tenant: result.tenant }; } @Post('logout') async logout(@CurrentUser() user: JwtPayload, @Req() req: Request, @Res(...) res: Response) { await this.authService.logout(user.sub, req.cookies?.[COOKIE_NAMES.REFRESH_TOKEN]); this.clearAuthCookies(res); // Limpa cookies } }

4. Serviço de Criptografia (AES-256-GCM)

📁 apps/api/src/common/security/encryption.service.ts

import * as crypto from 'crypto'; @Injectable() export class EncryptionService implements OnModuleInit { private readonly algorithm = 'aes-256-gcm'; private encryptionKey: Buffer | null = null; onModuleInit() { const keyHex = this.configService.get('ENCRYPTION_KEY'); if (keyHex?.length === 64) { this.encryptionKey = Buffer.from(keyHex, 'hex'); } } /** * Criptografa dados sensíveis * @returns Base64: iv + authTag + ciphertext */ encrypt(plaintext: string): string { const iv = crypto.randomBytes(16); // IV aleatório por operação const cipher = crypto.createCipheriv(this.algorithm, this.encryptionKey, iv); const encrypted = Buffer.concat([cipher.update(plaintext, 'utf8'), cipher.final()]); const authTag = cipher.getAuthTag(); return Buffer.concat([iv, authTag, encrypted]).toString('base64'); } /** * Hash com HMAC-SHA-256 (protege contra rainbow tables) */ hash(data: string): string { const pepper = this.configService.get('HASH_PEPPER') || this.encryptionKey?.toString('hex').slice(0, 32); return crypto.createHmac('sha256', pepper).update(data).digest('hex'); } }

@Injectable() export class SecretsService implements OnModuleInit { private secrets: Map<string, string> = new Map(); private readonly useVault: boolean; constructor(private readonly configService: ConfigService) { this.useVault = !!this.configService.get('INFISICAL_TOKEN'); } async onModuleInit() { if (this.useVault) await this.loadFromVault(); } /** Prioridade: Vault > Environment Variable */ get(key: string): string | undefined { return this.secrets.get(key) || this.configService.get(key); } private async loadFromVault(): Promise<void> { const { InfisicalClient } = await import('@infisical/sdk'); const client = new InfisicalClient({ siteUrl: 'https://app.infisical.com' }); const secretsList = await client.listSecrets({ environment: 'prod', projectId: this.configService.get('INFISICAL_PROJECT_ID'), }); for (const secret of secretsList) { this.secrets.set(secret.secretKey, secret.secretValue); } } }

1. Armazenando API Keys de Terceiros (Service)

📁 apps/api/src/modules/integrations/integrations.service.ts

import { Injectable } from '@nestjs/common'; import { PrismaService } from '../../common/database/prisma.service'; import { EncryptionService } from '../../common/security'; @Injectable() export class IntegrationsService { constructor( private readonly prisma: PrismaService, private readonly encryption: EncryptionService, ) {} /** * Salva API key de integração (ex: Stripe, SendGrid) * A chave é criptografada antes de ir pro banco */ async saveApiKey(tenantId: string, provider: string, apiKey: string) { // Criptografa a API key antes de salvar const encryptedKey = this.encryption.encrypt(apiKey); return this.prisma.integration.upsert({ where: { tenantId_provider: { tenantId, provider } }, create: { tenantId, provider, apiKey: encryptedKey }, update: { apiKey: encryptedKey }, }); } /** * Recupera API key descriptografada para uso */ async getApiKey(tenantId: string, provider: string): Promise<string | null> { const integration = await this.prisma.integration.findUnique({ where: { tenantId_provider: { tenantId, provider } }, }); if (!integration) return null; return this.encryption.decrypt(integration.apiKey); } }

2. CPF/CNPJ com Busca Criptografada

📁 apps/api/src/modules/customers/customers.service.ts

@Injectable() export class CustomersService { constructor( private readonly prisma: PrismaService, private readonly encryption: EncryptionService, ) {} /** * Cria cliente com CPF criptografado + hash para busca */ async create(tenantId: string, dto: CreateCustomerDto) { // CPF normalizado (apenas números) const cpfNormalized = dto.cpf.replace(/\D/g, ''); return this.prisma.customer.create({ data: { tenantId, name: dto.name, email: dto.email, // CPF criptografado (para exibição) cpfEncrypted: this.encryption.encrypt(cpfNormalized), // Hash do CPF (para busca - mesmo input = mesmo hash) cpfHash: this.encryption.hash(cpfNormalized), }, }); } /** * Busca cliente por CPF usando o hash */ async findByCpf(tenantId: string, cpf: string) { const cpfHash = this.encryption.hash(cpf.replace(/\D/g, '')); const customer = await this.prisma.customer.findFirst({ where: { tenantId, cpfHash }, }); if (customer) { // Descriptografa CPF para retornar customer.cpf = this.encryption.decrypt(customer.cpfEncrypted); } return customer; } }
💡 Padrão Hash + Encrypt:
• Hash (HMAC-SHA-256): Permite buscar no banco sem expor o dado real
• Encrypt (AES-256-GCM): Armazena dado reversível para exibição
